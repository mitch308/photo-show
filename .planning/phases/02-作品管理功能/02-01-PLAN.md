@Entity('works')
export class Work {
  @PrimaryGeneratedColumn('uuid')
  id: string;

  @Column({ type: 'varchar', length: 255 })
  title: string;

  @Column({ type: 'text', nullable: true })
  description: string;

  @Column({ type: 'varchar', length: 500 })
  filePath: string;           // Relative path: uploads/works/2026-03/uuid.jpg

  @Column({ type: 'varchar', length: 500, nullable: true })
  thumbnailSmall: string;     // 300px thumbnail

  @Column({ type: 'varchar', length: 500, nullable: true })
  thumbnailLarge: string;     // 1200px thumbnail

  @Column({ type: 'varchar', length: 255 })
  originalFilename: string;   // Original uploaded filename

  @Column({ type: 'varchar', length: 50 })
  fileType: string;           // 'image' or 'video'

  @Column({ type: 'varchar', length: 20 })
  mimeType: string;           // 'image/jpeg', 'video/mp4', etc.

  @Column({ type: 'int', default: 0 })
  fileSize: number;           // In bytes

  @Column({ type: 'int', default: 0 })
  position: number;           // For drag-drop sorting

  @Column({ type: 'boolean', default: false })
  isPinned: boolean;          // Per D-18

  @Column({ type: 'boolean', default: true })
  isPublic: boolean;          // Per WORK-11

  @Column({ type: 'int', default: 0 })
  viewCount: number;

  @Column({ type: 'int', default: 0 })
  downloadCount: number;

  // Relationships
  @ManyToMany(() => Album, album => album.works)
  albums: Album[];

  @ManyToMany(() => Tag, tag => tag.works)
  tags: Tag[];

  @CreateDateColumn({ type: 'datetime', name: 'created_at' })
  createdAt: Date;

  @UpdateDateColumn({ type: 'datetime', name: 'updated_at' })
  updatedAt: Date;
}
```

import { Entity, PrimaryGeneratedColumn, ManyToOne, JoinColumn } from 'typeorm';
import { Work } from './Work.js';
import { Album } from './Album.js';

@Entity('work_albums')
export class WorkAlbum {
  @PrimaryGeneratedColumn('uuid')
  id: string;

  @ManyToOne(() => Work)
  @JoinColumn({ name: 'work_id' })
  work: Work;

  @ManyToOne(() => Album)
  @JoinColumn({ name: 'album_id' })
  album: Album;
}
```

import { Entity, PrimaryGeneratedColumn, ManyToOne, JoinColumn } from 'typeorm';

export class WorkTag {
